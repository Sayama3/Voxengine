cmake_minimum_required(VERSION 3.10)

project(glad VERSION 0.1.36 DESCRIPTION "OpenGL loader generated by glad 0.1.36 on Thu May 25 17:09:42 2023.")

include(GNUInstallDirs)

add_library(glad
        src/glad.c
        )


set_target_properties(glad PROPERTIES VERSION ${PROJECT_VERSION})
set_target_properties(glad PROPERTIES SOVERSION 0)

target_include_directories(glad PUBLIC include)

set_property(TARGET glad PROPERTY POSITION_INDEPENDENT_CODE ON)
